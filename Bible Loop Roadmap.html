// ── Airtable fetch ──

async function fetchTable(tableName, fields) {

  const params = new URLSearchParams({ pageSize: 100 });

  if (fields) fields.forEach(f => params.append('fields[]', f));

  const rows = [];

  let offset = null;

  do {

    if (offset) params.set('offset', offset);

    const res = await fetch(

      `https://api.airtable.com/v0/${BASE_ID}/${encodeURIComponent(tableName)}?${params}`,

      { headers: { Authorization: `Bearer ${AIRTABLE_TOKEN}` } }

    );

    if (!res.ok) throw new Error(`Airtable error ${res.status} on table "${tableName}"`);

    const data = await res.json();

    rows.push(...data.records);

    offset = data.offset || null;

  } while (offset);

  return rows;

}

// ── Helpers ──

function byId(arr, id) { return arr.find(x => x.id === id); }


function getConnected(sel) {

  const none = { outcomes:new Set(), caps:new Set(), objs:new Set(), feats:new Set() };

  if (!sel) return none;

  let outcomes=new Set(), caps=new Set(), objs=new Set(), feats=new Set();

  const type = sel.startsWith('rec') ? 'rec' : sel[0];


  const addCap = cid => {

    const c = byId(DB.caps, cid); if (!c) return;

    caps.add(cid);

    c.outcomeIds.forEach(o => outcomes.add(o));

    DB.objs.filter(l=>l.capIds.includes(cid)).forEach(l => {

      objs.add(l.id);

      DB.features.filter(f=>f.objIds.includes(l.id)).forEach(f=>feats.add(f.id));

    });

  };

  const addObj = lid => {

    const l = byId(DB.objs, lid); if (!l) return;

    objs.add(lid);

    l.capIds.forEach(c=>addCap(c));

    DB.features.filter(f=>f.objIds.includes(lid)).forEach(f=>feats.add(f.id));

  };


  // Determine which collection sel belongs to

  const isOutcome = DB.outcomes.some(x=>x.id===sel);

  const isCap     = DB.caps.some(x=>x.id===sel);

  const isObj     = DB.objs.some(x=>x.id===sel);

  const isFeat    = DB.features.some(x=>x.id===sel);


  if (isOutcome) {

    outcomes.add(sel);

    DB.caps.filter(c=>c.outcomeIds.includes(sel)).forEach(c=>addCap(c.id));

  } else if (isCap) {

    addCap(sel);

  } else if (isObj) {

    addObj(sel);

  } else if (isFeat) {

    feats.add(sel);

    const f = byId(DB.features, sel);

    f.objIds.forEach(o=>addObj(o));

    f.capIds.forEach(c=>addCap(c));

    f.outcomeIds.forEach(o=>outcomes.add(o));

  }

  return { outcomes, caps, objs, feats };

}


// ── TRACEABILITY MAP ──

let traceSel = null;

function renderTraceability() {

  const conn = getConnected(traceSel);

  const cols = [

    { title:'Outcomes',            items: DB.outcomes,  color: C.teal,   type:'o',

      label: x=>x.label, isDim: x=> traceSel && !conn.outcomes.has(x.id), isActive: x=>traceSel===x.id,

      nodeColor: x=>x.color },

    { title:'Core capabilities',   items: DB.caps,      color: C.purple, type:'c',

      label: x=>x.label, isDim: x=> traceSel && !conn.caps.has(x.id),     isActive: x=>traceSel===x.id,

      nodeColor: ()=>C.gray },

    { title:'Learning objectives',  items: DB.objs,      color: C.coral,  type:'l',

      label: x=>x.label, isDim: x=> traceSel && !conn.objs.has(x.id),     isActive: x=>traceSel===x.id,

      nodeColor: ()=>C.gray },

    { title:'Features',            items: DB.features,  color: C.pink,   type:'f',

      label: x=>x.label, isDim: x=> traceSel && !conn.feats.has(x.id),    isActive: x=>traceSel===x.id,

      nodeColor: ()=>C.gray },

  ];


  const colsHtml = cols.map(col => {

    const nodes = col.items.map(item => {

      const nc = col.nodeColor(item);

      const dim = col.isDim(item);

      const active = col.isActive(item);

      const qBadge = item.quarter

        ? `<span class="tag" style="background:#F1EFE8;color:#5F5E5A;">${item.quarter}</span>` : '';

      return `<div class="trace-node ${active?'active':''} ${dim?'dimmed':''}"

        style="background:${nc.bg};border-color:${dim?nc.mid+'44':nc.border};color:${nc.text};

        ${active?`box-shadow:0 0 0 2.5px ${nc.border}50`:''}"

        onclick="traceClick('${item.id}')">

        ${col.label(item)} ${qBadge}

      </div>`;

    }).join('');

    return `<div class="trace-col">

      <div class="trace-col-header" style="background:${col.color.bg};border-color:${col.color.border};color:${col.color.text}">${col.title}</div>

      <div class="trace-col-body" style="border-color:${col.color.border}">${nodes}</div>

    </div>`;

  }).join('');


  const countHtml = traceSel

    ? `<div class="trace-count">${conn.outcomes.size} outcome${conn.outcomes.size!==1?'s':''} &middot; ${conn.caps.size} capabilit${conn.caps.size!==1?'ies':'y'} &middot; ${conn.objs.size} objective${conn.objs.size!==1?'s':''} &middot; ${conn.feats.size} feature${conn.feats.size!==1?'s':''}</div>` : '';


  document.getElementById('tab-traceability').innerHTML = `

    <div style="font-size:11px;color:#888780;margin-bottom:12px;text-align:center">Click any node to trace connections. Click again to reset.</div>

    <div class="trace-grid">${colsHtml}</div>

    ${countHtml}`;

}


function traceClick(id) {

  traceSel = traceSel === id ? null : id;

  renderTraceability();

}

let execSelOutcome = null, execSelFeature = null;

function renderExec() {

  const quarters = [...new Set(DB.features.map(f=>f.quarter).filter(Boolean))].sort();

  const visFeatures = DB.features.filter(f => !execSelOutcome || f.outcomeIds.includes(execSelOutcome));


  const outcomeCards = DB.outcomes.map(o => {

    const dim = execSelOutcome && execSelOutcome !== o.id;

    return `<div class="outcome-card" onclick="execOutcomeClick('${o.id}')"

      style="background:${dim?'#F1EFE8':o.color.bg};border-color:${dim?'#D3D1C7':o.color.border};opacity:${dim?0.45:1}">

      <h3 style="color:${dim?'#888780':o.color.text}">${o.label}</h3>

      <p style="color:${dim?'#B4B2A9':o.color.text}">${o.desc}</p>

    </div>`;

  }).join('');


  const timelineHtml = quarters.map(q => {

    const qf = visFeatures.filter(f=>f.quarter===q);

    const cards = qf.length === 0

      ? `<span style="font-size:11px;color:#b4b2a9;padding-top:8px;">No features match this outcome</span>`

      : qf.map(f => {

          const ss = statusStyle(f.status);

          const dots = f.outcomeIds.map(oid => {

            const o = byId(DB.outcomes, oid);

            return o ? `<span class="dot" style="background:${o.color.mid}" title="${o.label}"></span>` : '';

          }).join('');

          return `<div class="feature-card ${execSelFeature===f.id?'selected':''}" onclick="execFeatureClick('${f.id}')">

            <h4>${f.label}</h4>

            <div class="meta">

              <span class="tag" style="background:${ss.bg};color:${ss.text}">${f.status}</span>

              ${dots}

            </div>

          </div>`;

        }).join('');

    return `<div class="timeline-row">

      <div class="q-label">${q}</div>

      <div class="q-features">${cards}</div>

    </div>`;

  }).join('');


  let detailHtml = '';

  if (execSelFeature) {

    const f = byId(DB.features, execSelFeature);

    if (f) {

      const ss = statusStyle(f.status);

      const outcomePills = f.outcomeIds.map(oid => {

        const o = byId(DB.outcomes, oid);

        return o ? `<span class="pill" style="background:${o.color.bg};color:${o.color.text};border:1px solid ${o.color.border}">${o.label}</span>` : '';

      }).join('');

      detailHtml = `<div class="detail-panel">

        <div style="display:flex;justify-content:space-between;align-items:flex-start;margin-bottom:8px">

          <h3>${f.label}</h3>

          <span class="tag" style="background:${ss.bg};color:${ss.text};margin-left:8px">${f.status}</span>

        </div>

        ${f.problem ? `<div style="font-size:12px;color:#444441;line-height:1.5;margin-bottom:10px;padding:8px 10px;background:#fff;border:1px solid #e0dfd8;border-radius:6px">${f.problem}</div>` : ''}

        <div class="detail-section"><div class="label">Outcomes</div><div class="items">${outcomePills}</div></div>

      </div>`;

    }

  }


  const filterNote = execSelOutcome

    ? `<span style="color:${byId(DB.outcomes,execSelOutcome)?.color.mid}"> · filtered by "${byId(DB.outcomes,execSelOutcome)?.label}"</span>` : '';


  document.getElementById('tab-exec').innerHTML = `

    <div style="font-size:11px;font-weight:500;text-transform:uppercase;letter-spacing:.07em;color:#888780;margin-bottom:8px">Strategic outcomes</div>

    <div class="outcome-row">${outcomeCards}</div>

    <div style="font-size:11px;font-weight:500;text-transform:uppercase;letter-spacing:.07em;color:#888780;margin-bottom:10px">

      Feature timeline${filterNote}

    </div>

    ${timelineHtml}

    ${detailHtml}

    <div style="margin-top:18px;display:flex;gap:12px;flex-wrap:wrap">

      ${Object.entries(STATUS_STYLE).map(([s,c])=>`<div style="display:flex;align-items:center;gap:5px"><span style="width:9px;height:9px;border-radius:2px;background:${c.bg};border:1px solid ${c.text};display:inline-block"></span><span style="font-size:11px;color:#888780">${s}</span></div>`).join('')}

    </div>`;

}

}

let partnerSel = null, partnerLayer = 'outcomes';

function renderPartner() {

  const quarters = [...new Set(DB.features.map(f=>f.quarter).filter(Boolean))].sort();


  const layerBtns = ['outcomes','capabilities','objectives'].map(l =>

    `<button class="layer-btn ${partnerLayer===l?'active':''}" onclick="partnerLayer='${l}';renderPartner()">${

      l==='outcomes'?'Outcomes':l==='capabilities'?'+ Capabilities':'+ Objectives'

    }</button>`

  ).join('');


  let stratHtml = '';

  if (partnerLayer === 'outcomes') {

    stratHtml = `<div class="outcome-row">${DB.outcomes.map(o=>`

      <div class="outcome-card" style="background:${o.color.bg};border-color:${o.color.border};cursor:default">

        <h3 style="color:${o.color.text}">${o.label}</h3>

        <p style="color:${o.color.text}">${o.desc}</p>

      </div>`).join('')}</div>`;

  } else {

    stratHtml = DB.outcomes.map(o => {

      const oCaps = DB.caps.filter(c=>c.outcomeIds.includes(o.id));

      const capsHtml = oCaps.map(c => {

        const objsHtml = partnerLayer==='objectives'

          ? `<ul style="margin:4px 0 0;padding-left:14px">${DB.objs.filter(l=>l.capIds.includes(c.id)).map(l=>`<li style="font-size:10.5px;color:#5F5E5A;line-height:1.5">${l.label}</li>`).join('')}</ul>` : '';

        return `<div class="card-sm" style="min-width:150px;flex:1">

          <div style="font-size:11px;font-weight:500;color:#1a1a18">${c.label}</div>${objsHtml}</div>`;

      }).join('');

      return `<div style="border-radius:8px;border:1px solid ${o.color.border};background:${o.color.bg};padding:10px 12px;margin-bottom:8px">

        <div style="font-size:12px;font-weight:500;color:${o.color.text};margin-bottom:7px">${o.label}</div>

        <div style="display:flex;flex-wrap:wrap;gap:6px">${capsHtml}</div>

      </div>`;

    }).join('');

  }


  const timelineHtml = quarters.map(q => {

    const cards = DB.features.filter(f=>f.quarter===q).map(f => {

      const ss = statusStyle(f.status);

      const userFacing = !['infrastructure','backend'].some(k=>(f.type||'').toLowerCase().includes(k));

      const dots = f.outcomeIds.map(oid => {

        const o = byId(DB.outcomes, oid);

        return o ? `<span class="dot" style="background:${o.color.mid}"></span>` : '';

      }).join('');

      return `<div class="feature-card ${partnerSel===f.id?'selected':''}" onclick="partnerFeatureClick('${f.id}')">

        <h4>${f.label}</h4>

        <div class="meta">

          <span class="tag" style="background:${ss.bg};color:${ss.text}">${f.status}</span>

          <span class="tag" style="background:${userFacing?'#E6F1FB':'#F1EFE8'};color:${userFacing?'#0C447C':'#5F5E5A'}">${userFacing?'User-facing':'Infrastructure'}</span>

          ${dots}

        </div>

      </div>`;

    }).join('');

    return `<div class="timeline-row"><div class="q-label">${q}</div><div class="q-features">${cards}</div></div>`;

  }).join('');


  let detailHtml = '';

  if (partnerSel) {

    const f = byId(DB.features, partnerSel);

    if (f) {

      const ss = statusStyle(f.status);

      const opills = f.outcomeIds.map(oid=>{const o=byId(DB.outcomes,oid);return o?`<span class="pill" style="background:${o.color.bg};color:${o.color.text};border:1px solid ${o.color.border}">${o.label}</span>`:''}).join('');

      const cpills = f.capIds.map(cid=>{const c=byId(DB.caps,cid);return c?`<span class="pill" style="background:${C.gray.bg};color:${C.gray.text};border:1px solid ${C.gray.border}">${c.label}</span>`:''}).join('');

      const lpills = f.objIds.map(lid=>{const l=byId(DB.objs,lid);return l?`<span class="pill" style="background:${C.gray.bg};color:${C.gray.text};border:1px solid ${C.gray.border}">${l.label}</span>`:''}).join('');

      detailHtml = `<div class="detail-panel">

        <div style="display:flex;justify-content:space-between;align-items:flex-start;margin-bottom:8px">

          <h3>${f.label}</h3>

          <span class="tag" style="background:${ss.bg};color:${ss.text};margin-left:8px">${f.status}</span>

        </div>

        ${f.problem?`<div style="font-size:12px;line-height:1.5;margin-bottom:10px;padding:8px 10px;background:#fff;border:1px solid #e0dfd8;border-radius:6px;color:#444441">${f.problem}</div>`:''}

        ${f.risks?`<div style="font-size:12px;line-height:1.5;margin-bottom:10px;padding:8px 10px;background:#FAECE7;border:1px solid #F5C4B3;border-radius:6px;color:#712B13"><strong>Risks:</strong> ${f.risks}</div>`:''}

        <div class="detail-section"><div class="label">Outcomes</div><div class="items">${oills||'—'}</div></div>

        <div class="detail-section"><div class="label">Core capabilities</div><div class="items">${cpills||'—'}</div></div>

        <div class="detail-section"><div class="label">Learning objectives</div><div class="items">${lpills||'—'}</div></div>

      </div>`;

    }

  }


  document.getElementById('tab-partner').innerHTML = `

    <div class="card" style="margin-bottom:18px">

      <div style="font-size:13px;font-weight:500;margin-bottom:4px">Bible Loop — partner roadmap</div>

      <div style="font-size:12px;color:#888780;line-height:1.5">This view shows how planned features connect to the outcomes and learning goals we're building toward together. Use the layer toggle to explore the strategic logic, and click any feature for detail.</div>

    </div>

    <div class="layer-toggle"><span>Show strategy layer:</span>${layerBtns}</div>

    ${stratHtml}

    <div style="font-size:11px;font-weight:500;text-transform:uppercase;letter-spacing:.07em;color:#888780;margin-bottom:10px;margin-top:4px">Delivery timeline</div>

    ${timelineHtml}

    ${detailHtml}`;

}


function partnerFeatureClick(id) {

  partnerSel = partnerSel===id ? null : id;

  renderPartner();

}
